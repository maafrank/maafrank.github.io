# Medical imaging CNN pipeline
import tensorflow as tf
from tensorflow.keras.applications import ResNet50

# Load pre-trained model
base_model = ResNet50(weights='imagenet', include_top=False)

# Add custom layers for medical diagnosis
model = tf.keras.Sequential([
    base_model,
    tf.keras.layers.GlobalAveragePooling2D(),
    tf.keras.layers.Dense(128, activation='relu'),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(2, activation='softmax')  # cancer/no-cancer
])

# Fine-tune with medical data
model.compile(optimizer='adam', 
             loss='categorical_crossentropy',
             metrics=['accuracy'])

Recommendation Systems: The Personalization Engine

Perhaps the most financially successful AI application, recommendation systems drive engagement and revenue:

# Netflix-style recommendation system
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

class RecommendationEngine:
    def __init__(self, user_item_matrix):
        self.matrix = user_item_matrix
        
    def collaborative_filtering(self, user_id, n_recommendations=10):
        # Calculate user similarities
        user_similarities = cosine_similarity(self.matrix)
        
        # Find similar users
        similar_users = user_similarities[user_id].argsort()[:-10:-1]
        
        # Generate recommendations
        recommendations = self._generate_recs(user_id, similar_users)
        return recommendations[:n_recommendations]

Techniques: Collaborative filtering, content-based filtering, deep learning
